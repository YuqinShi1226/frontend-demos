## 1. this

##### 1.1
```javascript
function show () {
  console.log('this:', this)
}
var obj = { show: show }
obj.show()
```
Result: `this: obj`
> 点操作符执行的函数，this是调用函数的对象即 `obj { show: fn }`

```javascript
function show () {
  console.log('this:', this)
}
var obj = {
  show: function () {
    show()
  }
}
obj.show()
```
Result: `this: Window`
> obj.show里面的 show是直接调用，所以this指向window

##### 1.2
```javascript
var obj = {
  show: function () {
    console.log('this:', this)
  }
}
(0, obj.show)()
```
Result: `this: Window`
> 括号操作符返回的是obj的show 函数，之后直接调用该函数，所以this指向window

##### 1.3
```javascript
var obj = {
  sub: {
    show: function () {
      console.log('this:', this);
    }
  }
}
obj.sub.show()
```
Result: `this: sub `
> 点操作符执行的函数，this是调用函数的对象即 `sub { show: fn }`

##### 1.4
```javascript
var obj = {
  show: function () {
    console.log('this:', this);
  }
}
var newObj = new obj.show()
```
Result: `this: newObj`
> 构造函数实例，函数this指向该实例

##### 1.5
```javascript
var obj = {
  show: function () {
    console.log('this:', this);
  }
}
var newObj = new (obj.show.bind(obj))()
```
Result: `this: newObj`
> bind将obj.show的this绑定到obj，之后用 new 构造返回的函数实例，this指向该实例

##### 1.6
```javascript
var obj = {
  show: function () {
    console.log('this:', this);
  }
}
var elem = document.getElementById('book-search-results')
// 1
elem.addEventListener('click', obj.show)
// 2
elem.addEventListener('click', obj.show.bind(obj))
// 3
elem.addEventListener('click', function () {
  obj.show()
})
```
Result:
```
this: elem
this: obj
this: obj
```
> 1. 第一个 `obj.show` 返回函数作为回调方法，之后 elem 点击时，elem 作为 this 调用这个回调方法，所以 this 指向 elem
> 2. 第二个this 被 bind 到 obj，所以 this 指向obj
> 3. 点操作符，this 指向调用者 obj

## 2. scope

##### 2.1
```javascript
var person = 1
function showPerson () {
  var person = 2
  console.log(person)
}
showPerson()
```
Result: `2`
> 首先从函数内部寻找变量，内部存在变量person，赋值为 2

##### 2.2
```javascript
var person = 1

showPerson()
```
Result: `undefined`
> 首先从函数内部寻找变量，内部存在变量 person，console.log 在person 赋值之前，所以 person 为 undefined

> 函数

##### 2.5
```javascript
// 1
for(var i = 0; i < 10; i++) {
  console.log(i)
}
// 2
for(var i = 0; i < 10; i++) {
  setTimeout(function () {
    console.log(i)
  }, 0);
}
// 3
for(var i = 0; i < 10; i++) {
  (function (i) {
    setTimeout(function () {
      console.log(i)
    }, 0);
  })(i)
}
// 4
for(let i = 0; i < 10; i++) {
  console.log(i)
}

```
Result:
```
0, 1 ... 9          // 1st fun
0, 1 ... 9          // 4th fun
10, 10, 10 ... 10   // 2nd fun
0, 1 ... 9          // 3rd fun
```
>执行顺序为 1, 4, 2, 3, 因为 2，3中包含了timeout，所以回调函数会被推到 task queue, 等待当前脚本执行完毕才会执行。
> 1. 首先执行第一个循环，依次打印 0-9
> 2. 第2个循环会产生 10个 task， 被推到task queue中，循环之后，i的值为10，而回调函数的i指向的是循环里面的i，此时i的值为10。所以会打印 10 个 10。
> 3. 循环内部的匿名函数接收了 i 作为入参，之后保留到内部，之后setimeout 回调中的 i 使用的是匿名函数的 i, 所以是 0 - 9
> 4. 同 1

## 3. OOP

##### 3.1
```javascript
function Person () {
  this.name = 1
  return {}
}

var person = new Person()

console.log('name:', person.name)
```
Result: `undefined`
> 因为Person return 了一个空的对象，所以person.name 为 undefined

##### 3.2
```javascript
function Person () {
  this.name = 1
}

Person.prototype = {
  name: 2,
  show: function () {
    console.log('name is:', this.name)
  }
}
var person = new Person()

person.show()
```
Result: `name is:1`
> 因为在 Person 中定义了name，person 是 Person 的实例，指向Person的contructor，name 为 1，覆盖了原型的 2

##### 3.3
```javascript
function Person () {
  this.name = 1
}

Person.prototype = {
  name: 2,
  show: function () {
    console.log('name is:', this.name)
  }
}
var person = new Person()

Person.prototype.show = function () {
  console.log('new show')
}

person.show()
```
Result: `new show`
> person 是 Person 的实例，person 和 Person 都没有 show 方法，沿着原型链找，Person的原型的show 最后定义为打印为 `new show`


##### 3.4
```javascript
function Person () {
  this.name = 1
}

Person.prototype = {
  name: 2,
  show: function () {
    console.log('name is:', this.name)
  }
}
var person = new Person()
var person2 = new Person()

person.show = function () {
  console.log('new show')
}

person2.show()
person.show()
```
Result:
```
name is:1
new show
```
> - person 是 Person 的实例，person 和 Person 都没有 show 方法，沿着原型链找，Person的原型的 show 函数 `name is: 1`
> - person2 是 Person 的实例， person 的 show 函数定义为 `new show`

## 4
```javascript
function Person () {
  this.name = 1
}

Person.prototype = {
  name: 2,
  show: function () {
    console.log('name is:', this.name)
  }
}

Person.prototype.show()
(new Person()).show()

```
Result:
```
name is:2
name is:1
```
> 1. 直接调用 Person 原型的show，this 指向Person 原型，name = 2
> 2. new 操作符会创建一个新的Person的实例，this指向Person的构造函数，name = 1
